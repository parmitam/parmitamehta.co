CoffeeScript to JavaScript
coffeescript-to-javascript.js.coffee 
# Type here!

math =
  root:   Math.sqrt
  square: square
  cube:   (x) -> x * square x

alert "Three cubed is #{math.cube 3}"
 coffeescript-to-javascript.js 
(function() {
  var math;

  math = {
    root: Math.sqrt,
    square: square,
    cube: function(x) {
      return x * square(x);
    }
  };

  alert("Three cubed is " + (math.cube(3)));

}).call(this);
